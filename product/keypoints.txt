允许 Docker 容器访问宿主机服务

# 为 Docker 添加访问宿主机的路由
echo '{
  "dns": ["8.8.8.8"],
  "features": {"buildkit": true},
  "experimental": true,
  "hosts": ["tcp://0.0.0.0:2375", "unix:///var/run/docker.sock"]
}' | sudo tee /etc/docker/daemon.json


sudo systemctl restart docker



开放防火墙端口
sudo ufw allow 3306/tcp
sudo ufw reload


修改模型后的标准流程
# 1. 生成迁移脚本
flask db migrate -m "Add description to products"

# 2. 检查生成的迁移脚本
# 文件位置: migrations/versions/xxx_add_description_to_products.py

# 3. 应用迁移
flask db upgrade
回滚迁移
# 查看历史版本
flask db history

# 回滚到指定版本
flask db downgrade <revision_id>
